function onResults(results) { // 1. 將清晰原圖畫在底層 canvasCtx.globalCompositeOperation = 'source-over'; canvasCtx.filter = 'none'; canvasCtx.drawImage(results.image, 0, 0, canvasElement.width, canvasElement.height); if (results.multiFaceLandmarks && results.multiFaceLandmarks.length > 0) { const landmarks = results.multiFaceLandmarks[0]; const nose = landmarks[1]; const x = nose.x * canvasElement.width; const y = nose.y * canvasElement.height; // 2. 建立離線畫布來處理美肌圖層 (避免直接污染主畫面) const tempCanvas = document.createElement('canvas'); tempCanvas.width = canvasElement.width; tempCanvas.height = canvasElement.height; const tempCtx = tempCanvas.getContext('2d'); // 3. 在離線畫布上套用美肌濾鏡 const blurVal = blurRange.value; const brightVal = brightRange.value; tempCtx.filter = `blur(${blurVal}px) brightness(${brightVal})`; tempCtx.drawImage(results.image, 0, 0, tempCanvas.width, tempCanvas.height); // 4. 製作漸層羽化遮罩 (核心修正) // 使用 destination-in，讓漸層透明度決定美肌圖層的顯示範圍 tempCtx.globalCompositeOperation = 'destination-in'; // 設定放射狀漸層：中心全滿(1)，擴散到外圍變透明(0) const gradient = tempCtx.createRadialGradient(x, y, 40, x, y, 130); gradient.addColorStop(0, 'rgba(0, 0, 0, 1)'); gradient.addColorStop(1, 'rgba(0, 0, 0, 0)'); tempCtx.fillStyle = gradient; tempCtx.beginPath(); tempCtx.ellipse(x, y, 100, 140, 0, 0, 2 * Math.PI); tempCtx.fill(); // 5. 將羽化後的美肌圖層疊加回主畫面 canvasCtx.globalCompositeOperation = 'source-over'; canvasCtx.drawImage(tempCanvas, 0, 0); } }
